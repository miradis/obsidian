#SQL Общее количество число
```
select count(*)

from contract.company_requisites cr

where cr.created_at >= 'startDate' and cr.created_at <= 'endDate'
```
#SQL по месяцам
```
SELECT

DATE_TRUNC('month', cr.created_at) AS date,

COUNT(*)

FROM

contract.company_requisites cr

WHERE

cr.created_at >= 'startDate' AND

cr.created_at <= 'endDate'

GROUP BY

DATE_TRUNC('month', cr.created_at)

ORDER BY

date;
```
#SQL  по дням
```
SELECT

DATE_TRUNC('day', cr.created_at) AS date,

COUNT(*)

FROM

contract.company_requisites cr

WHERE

cr.created_at >= 'startDate' AND

cr.created_at <= 'endDate'

GROUP BY

DATE_TRUNC('day', cr.created_at)

ORDER BY

date;
```
